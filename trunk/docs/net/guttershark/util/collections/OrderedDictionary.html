Classpublic final class OrderedDictionary
InheritanceOrderedDictionary flash.utils.Proxy

OrderedDictionary provides a Dictionary like collection object which persists the order of entries similar to an Array. 
 
  var od:OrderedDictionary = new OrderedDictionary();
  od['test0'] = "foo";
  od['test1'] = "bar";
  od.push("test3","doe");
  
  trace(od['test1']);
  trace(od['test3']);
  
  trace(od.pop());
  
  var i:int = 0;
  var l:int = od.length;
  for(i;i<l;i++)
  { 
      trace(od[i]);
  }
